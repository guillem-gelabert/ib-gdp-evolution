```python
def fetch_eurostat_series(
    dataset: str,
    geos: list[str],
    unit: str,
    *,
    na_item: str = "",
    datalake_index: dict | None = None,
    datalake_root: Path | None = None,
    extra_params: str = "",
) -> pd.DataFrame:
    """Try data-lake artifact first, fall back to live Eurostat API (per D-09)."""
    if datalake_index and datalake_root:
        sorted_geos = sorted(geos)
        # Build lookup key matching index format
        if na_item:
            key = f"{dataset}|{unit}|{na_item}|{'|'.join(sorted_geos)}"
        elif len(sorted_geos) == 1:
            key = f"{dataset}|{unit}|{sorted_geos[0]}"
        else:
            key = f"{dataset}|{unit}|{'|'.join(sorted_geos)}"
        source_id = datalake_index.get(key)
        if source_id:
            raw_path = datalake_root / "lake" / "sources" / source_id / "raw.json"
            if raw_path.exists():
                d = json.loads(raw_path.read_text(encoding="utf-8"))
                df = _eurostat_json_to_df(d)
                out = df.rename(columns={"geo": "nuts2_code", "time": "year"})
                out["year"] = out["year"].astype(int)
                out = out[out["year"].between(CHAIN_START, YEAR_MAX)]
                if len(geos) > 1:
                    out = out[out["nuts2_code"].isin(geos)]
                return out[["nuts2_code", "year", "value"]].drop_duplicates(["nuts2_code", "year"])
    # Fallback: live Eurostat API
    geo_q = "&".join(f"geo={g}" for g in geos)
    params = f"unit={unit}&{geo_q}"
    if na_item:
        params = f"unit={unit}&na_item={na_item}&{geo_q}"
    if extra_params:
        params = f"{params}&{extra_params}"
    d = _eurostat_get(dataset, params)
    df = _eurostat_json_to_df(d)
    out = df.rename(columns={"geo": "nuts2_code", "time": "year"})
    out["year"] = out["year"].astype(int)
    out = out[out["year"].between(CHAIN_START, YEAR_MAX)]
    return out[["nuts2_code", "year", "value"]].drop_duplicates(["nuts2_code", "year"])
```
