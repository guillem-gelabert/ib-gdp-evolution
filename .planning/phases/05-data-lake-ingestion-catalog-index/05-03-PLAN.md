**1. Add `baseline_regression_check()` after `run_checks()` (around line 710):**

```python
def baseline_regression_check(
    new_csvs: dict[str, pd.DataFrame],
    workspace: Path,
) -> CheckResult:
    """Compare full-ETL output against local-proxy CSVs (growth-rate correlation, per D-10/VALID-03)."""
    correlations: list[tuple[str, float]] = []
    for slug, new_df in new_csvs.items():
        proxy_path = workspace / "public" / "data" / f"act2_{slug}.csv"
        if not proxy_path.exists():
            continue
        proxy = pd.read_csv(proxy_path)
        proxy["year"] = pd.to_numeric(proxy["year"], errors="coerce")
        proxy["gdp_pc"] = pd.to_numeric(proxy["gdp_pc"], errors="coerce")
        proxy = proxy.dropna(subset=["year", "gdp_pc"])
        new_years = set(new_df["year"].astype(int))
        proxy_years = set(proxy["year"].astype(int))
        common_years = sorted(new_years & proxy_years)
        if len(common_years) < 3:
            continue
        n = new_df[new_df["year"].isin(common_years)].sort_values("year")
        p = proxy[proxy["year"].isin(common_years)].sort_values("year")
        n_growth = n["gdp_pc_2011ppp"].pct_change().dropna()
        p_growth = p["gdp_pc"].pct_change().dropna()
        if len(n_growth) > 1 and len(p_growth) > 1:
            corr = float(n_growth.corr(p_growth))
            if not math.isnan(corr):
                correlations.append((slug, corr))
    if not correlations:
        return CheckResult("8) Baseline regression", "WARN", 1, ["No proxy CSVs found for comparison"])
    min_slug, min_corr = min(correlations, key=lambda x: x[1])
    avg_corr = sum(c for _, c in correlations) / len(correlations)
    details = [f"{slug}: corr={corr:.4f}" for slug, corr in correlations]
    if min_corr >= 0.95:
        return CheckResult("8) Baseline regression", "PASS", 0,
                           [f"min={min_slug}:{min_corr:.4f}, avg={avg_corr:.4f}"] + details)
    return CheckResult("8) Baseline regression", "WARN", 1,
                       [f"min={min_slug}:{min_corr:.4f} < 0.95, avg={avg_corr:.4f}"] + details)
```
